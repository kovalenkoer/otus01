Создаем каталоги:
mkdir /mnt/root_old
mkdir /mnt/root_new
mount /dev/sdb1 /mnt/root_old
xfsdump -f /mnt/root_old/root.dmp /

2.Выключаем ВМ, подключаем iso Centos 7
Power - Troble - Rescue - 3 (Skip to shell)
#
mkdir /mnt/backup

mount /dev/sda1 /mnt/backup #Диски сменили буквы

vgchange -ay  # активация VG
lvremove /dev/VolGroup00/LogVol00  # удаление LV
lvcreate -L 8G -n LogVol00 VolGroup00  # новый LV с тем же именем
vgchange -ay  
mkfs.xfs -L root /dev/VolGroup00/LogVol00  
mkdir /mnt/root
mount /dev/VolGroup00/LogVol00 /mnt/root  
xfsrestore -f /mnt/backup/root.dmp /mnt/root  # восстанавливаем LV из бекапа

umount /mnt/backup

parted:
Удаляем партиции (parted) sda1, sda
Конвертируем диск sda из GPT в MBR

vgextend VolGroup00 /dev/sdф  # добавляем в VG еще один диск
lvcreate -L 8G -m1 -n var VolGroup00 #зеркало
vgchange -ay
mkfs.xfs /dev/VolGroup00/var
mkdir /mnt/home  # для home
mkdir /mnt/var # для var
mount /dev/VolGroup00/var /mnt/var 
mv /mnt/root/var /mnt/var # перемещаем файлы


lvcreate -L 9G -n home VolGroup00  # создаем LV для home
vgchange -ay  
mkfs.ext4 /dev/VolGroup00/home  # форматируем LV в ext4
mount /dev/VolGroup00/home /mnt/home  # монтируем новый LV для home
mv /mnt/root/home/* /mnt/home/  # перемещаем файлы

vi /mnt/root/etc/fstab  # добавляем в fstab var и home монтирование

fstab:
  /dev/mapper/VolGroup00-var /var xfs defaults 0 0
  /dev/mapper/VolGroup00-home /home ext4 defaults 0 0

reboot
